Calculate the average number of goals per match, including home and away breakdowns

SELECT 
  AVG(FullTimeHomeGoals + FullTimeAwayGoals) AS avg_goals_per_match,
  AVG(FullTimeHomeGoals) AS avg_home_goals,
  AVG(FullTimeAwayGoals) AS avg_away_goals
FROM epl_matches
avg_goals_per_match	avg_home_goals	avg_away_goals

Calculate the distribution of match results (home wins, away wins, draws)

SELECT 
  Result,
  COUNT(*) AS match_count,
  ROUND(100.0 * COUNT(*) / SUM(COUNT(*)) OVER (), 2) AS percentage
FROM epl_matches
GROUP BY Result
ORDER BY match_count DESC

Away team win	2768	29.51
